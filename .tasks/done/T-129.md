Current detection logic in `src/main/ipc/taskTerminal.ts:157-167`:

```typescript
function tmuxParseAgentState(sessionName: string): AgentState {
  const capture = tmuxCapturePane(sessionName);

  if (!capture) return "idle";

  if (capture.includes("esc") && capture.includes("interrupt")) {
    return "active";
  }

  return "waiting";
}
```

Needs to become agent-aware:

```typescript
function tmuxParseAgentState(sessionName: string, agent: string): AgentState {
  const capture = tmuxCapturePane(sessionName);

  if (!capture) return "idle";

  if (agent === "copilot") {
    if (capture.includes("Esc") && capture.includes("cancel")) {
      return "active";
    }
  } else {
    if (capture.includes("esc") && capture.includes("interrupt")) {
      return "active";
    }
  }

  return "waiting";
}
```
